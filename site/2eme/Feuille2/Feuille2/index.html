def Lagrange(L,x,k):  #Lk(x) L=[x0,x1...xn]
    n=len(L)
    p=1
    for j in range (0,n+1):
        if j!=k:
            p=p*(x-L[j]/(L[k]-L[j]))
    return p

    z=0
    n=len(P) #calcul de la dimension du vecteur polynôme P
    for k in range (n-1, -1, -1):
